    ```typescript
    import { NextResponse } from "next/server"
    import { auth } from "@/lib/auth"
    import { db } from "@/lib/db"
    import { approvals, users, flows } from "../../../../../db/schema"
    import { and, eq, gte, lte, ilike, desc } from "drizzle-orm"

    function csvEscape(v: string): string {
      return `"${v.replace(/"/g, '""')}"`
    }
    function csvSafe(v: string): string {
      return /^[=+\-@]/.test(v) ? ` ${v}` : v
    }

    export async function GET(req: Request) {
      const session = await auth()
      if (!session?.user?.organizationId) {
        return NextResponse.json({ error: "Unauthorized" }, { status: 401 })
      }
      const orgId = session.user.organizationId
      const { searchParams: sp } = new URL(req.url)

      const conditions = [eq(approvals.organizationId, orgId)]   // T-02-03 mandatory
      const status = sp.get("status")
      if (status) conditions.push(eq(approvals.status, status as "pending"|"approved"|"rejected"))
      const flowId = sp.get("flowId")
      if (flowId) conditions.push(eq(approvals.flowId, flowId))
      const dateFrom = sp.get("dateFrom")
      if (dateFrom) {
        const d = new Date(dateFrom)
        if (!isNaN(d.getTime())) conditions.push(gte(approvals.createdAt, d))
      }
      const dateTo = sp.get("dateTo")
      if (dateTo) {
        const d = new Date(dateTo)
        if (!isNaN(d.getTime())) conditions.push(lte(approvals.createdAt, d))
      }
      const q = sp.get("q")
      if (q) conditions.push(ilike(approvals.title, `%${q}%`))

      const rows = await db.select({
        id: approvals.id,
        title: approvals.title,
        status: approvals.status,
        flowName: flows.name,
        createdAt: approvals.createdAt,
        resolvedByName: users.name,
        comment: approvals.comment,
      })
        .from(approvals)
        .leftJoin(users, eq(approvals.resolvedBy, users.id))
        .leftJoin(flows, eq(approvals.flowId, flows.id))
        .where(and(...conditions))
        .orderBy(desc(approvals.createdAt))

      const header = "ID,Título,Status,Fluxo,Data,Resolvido Por,Comentário\n"
      const body = rows.map(r => [
        r.id,
        csvEscape(csvSafe(r.title ?? "")),
        r.status,
        csvEscape(r.flowName ?? ""),
        r.createdAt.toISOString(),
        csvEscape(r.resolvedByName ?? ""),
        csvEscape(csvSafe(r.comment ?? "")),
      ].join(",")).join("\n")

      return new Response(header + body, {
        headers: {
          "Content-Type": "text/csv; charset=utf-8",
          "Content-Disposition": `attachment; filename="aprovacoes-${Date.now()}.csv"`,
        },
      })
    }
    ```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| client → /approvals page | authenticated user via NextAuth JWT |
| client searchParams → Drizzle WHERE | untrusted input (status, dates, query string) |
| client → /api/approvals/export | authenticated user; download triggered via `<a href download>` |
| CSV consumed by Excel/Sheets | external program (formula injection risk) |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-02-02 | Tampering / Information Disclosure | URL searchParams flowId / dateFrom / dateTo | mitigate | All Drizzle conditions ALWAYS lead with `eq(approvals.organizationId, orgId)` from session — flowId/status passed by user CANNOT cross tenant because outer AND clamps to orgId; dates validated with `isNaN(d.getTime())` |
| T-02-03 | Information Disclosure | /api/approvals/export missing orgId scoping | mitigate | Route re-applies same orgId condition first; integration test asserts cross-tenant rows absent; unit test asserts `eq(approvals.organizationId)` is invoked |
| T-02-03b | Information Disclosure | Unauthenticated CSV download | mitigate | `await auth()` returns 401 if no session.user.organizationId before any DB call |
| T-02-03c | Tampering | CSV formula injection in Excel | mitigate | `csvSafe()` prefixes `=+-@` with space; `csvEscape()` doubles quotes per RFC 4180 |
| T-02-03d | DoS | Unbounded export pulling huge result | accept | MVP: trial-tier orgs have low approval volume; add streaming + LIMIT in a future plan if needed |
</threat_model>
