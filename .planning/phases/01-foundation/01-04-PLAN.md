    ```typescript
    import { NextResponse } from "next/server"
    import { z } from "zod"
    import { auth } from "@/lib/auth"
    import {
      buildOrgLogoR2Key,
      presignUploadUrl,
      LOGO_ALLOWED_MIME_TYPES,
      MAX_LOGO_SIZE_BYTES,
    } from "@/lib/r2"

    const PRIVILEGED_ROLES = new Set(["owner", "admin"])
    const PRESIGN_TTL_MS = 10 * 60 * 1000

    const schema = z.object({
      filename: z.string().min(1).max(255),
      mimeType: z.string(),
      sizeBytes: z.number().int().positive().max(MAX_LOGO_SIZE_BYTES),
    })

    export async function POST(req: Request) {
      const session = await auth()
      if (!session?.user?.id) {
        return NextResponse.json({ error: "unauthorized" }, { status: 401 })
      }
      const orgId = session.user.organizationId
      if (!orgId) {
        return NextResponse.json({ error: "no_organization" }, { status: 400 })
      }
      if (!PRIVILEGED_ROLES.has(session.user.role ?? "")) {
        return NextResponse.json({ error: "forbidden" }, { status: 403 })
      }

      let body: unknown
      try { body = await req.json() } catch { return NextResponse.json({ error: "invalid_json" }, { status: 400 }) }
      const parsed = schema.safeParse(body)
      if (!parsed.success) {
        return NextResponse.json({ error: parsed.error.flatten() }, { status: 400 })
      }
      const { filename, mimeType, sizeBytes } = parsed.data

      if (!LOGO_ALLOWED_MIME_TYPES.has(mimeType)) {
        return NextResponse.json(
          { error: `mimeType não permitido. Aceitos: ${[...LOGO_ALLOWED_MIME_TYPES].join(", ")}` },
          { status: 422 },
        )
      }

      const r2Key = buildOrgLogoR2Key(orgId, filename)
      const expiresAt = new Date(Date.now() + PRESIGN_TTL_MS)
      const uploadUrl = await presignUploadUrl(r2Key, mimeType, sizeBytes)

      return NextResponse.json({ uploadUrl, r2Key, expiresAt: expiresAt.toISOString() })
    }
    ```

    ```typescript
    import { NextResponse } from "next/server"
    import { z } from "zod"
    import { eq } from "drizzle-orm"
    import { auth } from "@/lib/auth"
    import { db } from "@/lib/db"
    import { organizations } from "../../../../../db/schema"

    const PRIVILEGED_ROLES = new Set(["owner", "admin"])

    const schema = z
      .object({
        name: z.string().min(1).max(120).optional(),
        logo: z.union([z.string().min(1).max(512), z.null()]).optional(),
      })
      .refine((d) => d.name !== undefined || d.logo !== undefined, {
        message: "Pelo menos um campo (name ou logo) deve ser fornecido.",
      })

    export async function PATCH(req: Request) {
      const session = await auth()
      if (!session?.user?.id) {
        return NextResponse.json({ error: "unauthorized" }, { status: 401 })
      }
      const orgId = session.user.organizationId
      if (!orgId) {
        return NextResponse.json({ error: "no_organization" }, { status: 400 })
      }
      if (!PRIVILEGED_ROLES.has(session.user.role ?? "")) {
        return NextResponse.json({ error: "forbidden" }, { status: 403 })
      }

      let body: unknown
      try { body = await req.json() } catch { return NextResponse.json({ error: "invalid_json" }, { status: 400 }) }
      const parsed = schema.safeParse(body)
      if (!parsed.success) {
        return NextResponse.json({ error: parsed.error.flatten() }, { status: 400 })
      }

      const { name, logo } = parsed.data

      // Tenant guard: logo path must be in this org's namespace.
      if (typeof logo === "string") {
        const expectedPrefix = `org-logos/${orgId}/`
        if (!logo.startsWith(expectedPrefix)) {
          return NextResponse.json({ error: "invalid_logo_path" }, { status: 400 })
        }
      }

      const patch: Record<string, unknown> = {}
      if (name !== undefined) patch.name = name
      if (logo !== undefined) patch.logo = logo // can be null to clear

      await db.update(organizations).set(patch).where(eq(organizations.id, orgId))

      return NextResponse.json({ ok: true })
    }
    ```

       ```tsx
       interface OrgLogoProps {
       ```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Browser → /api/organizations/logo/presign | Untrusted file metadata (filename, mimeType, sizeBytes) crosses |
| Browser → R2 PUT (signed URL) | Browser uploads bytes directly to Cloudflare R2 — never traverses Vercel |
| Browser → /api/organizations/profile | Untrusted name + logo r2Key string crosses |
| Server → R2 read | Signed read URLs short-lived; signed by IAM credentials in env vars |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-04-01 | Tampering | Cross-tenant logo path injection (e.g., user submits `org-logos/OTHER_ORG/x.png` to PATCH /profile) | mitigate | PATCH route validates `logo.startsWith("org-logos/${orgId}/")` before update; rejects with 400. Verified by unit test. |
| T-04-02 | Spoofing | Non-privileged user calls presign or PATCH | mitigate | Both endpoints gate on `PRIVILEGED_ROLES.has(session.user.role)`. Returns 403 for member/viewer. |
| T-04-03 | Information Disclosure | Direct file upload to R2 with malicious mimeType (e.g., text/html for XSS) | mitigate | LOGO_ALLOWED_MIME_TYPES restricted to image/png, image/jpeg, image/webp ONLY. Server validates twice (Zod schema + set lookup). Client-side validation is convenience only — server is authoritative. |
| T-04-04 | Denial of Service | Logo upload with huge sizeBytes claim consumes server time | mitigate | Zod schema enforces sizeBytes <= MAX_LOGO_SIZE_BYTES (2MB) at API boundary. R2 presigned URL also encodes Content-Length-Range internally — oversized PUTs rejected by R2 itself. |
| T-04-05 | Elevation of Privilege | Stolen presigned URL reused by attacker | mitigate | Presigned URL TTL = 10min (PRESIGN_TTL_MS); after expiry, R2 rejects PUT. Risk window minimal; key path includes uuid so attacker cannot guess prior keys. |
| T-04-06 | Information Disclosure | Org logo URL leaks across tenants via Next/Image cache | mitigate | OrgLogo uses `unoptimized` prop; signed URL is fetched server-side per request and never cached by Next/Image. |
| T-04-07 | Repudiation | No audit log for name/logo changes | accept | Phase 1 scope. Future enhancement: write audit row on each PATCH. Acceptable for solo-tenant operator + small org count. |
| T-04-08 | Tampering | Stale logo R2 key after replacement (orphaned file) | accept | Old R2 object remains until lifecycle cleanup. Cost is negligible; explicit cleanup job is out of scope per plan boundary. |

**Blocking severity:** T-04-01 (cross-tenant injection), T-04-02 (role gate), T-04-03 (mimeType allowlist). All mitigated.
