    const hasUpstash =
      !!process.env.UPSTASH_REDIS_REST_URL && !!process.env.UPSTASH_REDIS_REST_TOKEN

    const redis = hasUpstash
      ? new Redis({
          url: process.env.UPSTASH_REDIS_REST_URL!,
          token: process.env.UPSTASH_REDIS_REST_TOKEN!,
        })
      : null

    type LimitResult = { success: boolean; limit: number; remaining: number; reset: number }

    function makeLimiter(
      limiter: ReturnType<typeof Ratelimit.slidingWindow> | ReturnType<typeof Ratelimit.fixedWindow>,
      prefix: string,
    ): { limit: (key: string) => Promise<LimitResult> } {
      if (!redis) {
        // Dev fallback when Upstash not configured — no-op
        return {
          limit: async () => ({ success: true, limit: 0, remaining: 0, reset: 0 }),
        }
      }
      return new Ratelimit({ redis, limiter, prefix })
    }

    // Replaces authRateMap (proxy.ts): 20 req/min per IP
    export const authRatelimit = makeLimiter(
      Ratelimit.slidingWindow(20, "60 s"),
      "pruma:auth",
    )

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Browser → NextAuth | Untrusted credentials cross into auth.ts JWT callback |
| App → Upstash REST | HTTP token traverses public internet (Upstash REST API) |
| App → Postgres | Local Docker socket (dev) / Neon HTTP TLS (prod) |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-01-01 | Spoofing | JWT emailVerified claim | mitigate | Claim derived server-side from `users.emailVerified IS NOT NULL`; never accepted from request body. Token signed by NextAuth secret. |
| T-01-02 | Tampering | emailOtpTokens.tokenHash | mitigate | bcrypt hash stored, raw OTP never persisted. `usedAt` flag prevents replay (enforced in Plan 02). |
| T-01-03 | Information Disclosure | UPSTASH_REDIS_REST_TOKEN in code | mitigate | Token read only via process.env; .env.example contains empty value; .env.local in .gitignore. |
| T-01-04 | Denial of Service | Cross-instance rate limit bypass | mitigate | Upstash Ratelimit replaces in-memory Map (proxy.ts). Atomic counter across all Vercel instances. INFRA-01. |
| T-01-05 | Elevation of Privilege | JWT emailVerified=true bypass | mitigate | Plan 02 enforces fresh signIn after verify-otp; no client can flip the claim without DB update + new JWT issuance. |
| T-01-06 | Repudiation | OTP attempt logging | accept | OTP attempts not audit-logged in Phase 1; brute-force capped via Upstash 5/15min ratelimiter (T-01-04). Acceptable risk: solo-tenant operator can reconstruct from Upstash logs if needed. |

**Blocking severity:** T-01-01 (auth claim spoofing), T-01-02 (OTP replay), T-01-04 (DoS). All mitigated above.
</threat_model>
