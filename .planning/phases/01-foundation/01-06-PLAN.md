       import { auth } from "@/lib/auth"
       import { NextResponse } from "next/server"
       import { authRatelimit, billingRatelimit } from "@/lib/ratelimit"

       const GUARDED_PREFIXES = ["/dashboard", "/flows", "/approvals", "/settings"]
       const BLOCKED_STATUSES = new Set(["canceled", "inactive"])

       // emailVerified gate (D-01) — paths reachable while unverified
       const EMAIL_VERIFY_BYPASS = new Set([
         "/verify-email",
         "/api/auth/verify-otp",
         "/api/auth/resend-otp",
         "/api/auth/signout",
       ])

       function isEmailVerifyBypass(pathname: string): boolean {
         if (EMAIL_VERIFY_BYPASS.has(pathname)) return true
         // Allow all NextAuth provider paths (signin, callback, csrf, session) so the user
         // can re-authenticate or sign out while unverified.
         if (pathname.startsWith("/api/auth/") &&
             !pathname.startsWith("/api/auth/onboarding")) {
           // Note: /api/auth/onboarding is treated by the existing onboarding guard, not here.
           return true
         }
         return false
       }

       export default auth(async (req) => {
         const { pathname } = req.nextUrl
         const session = req.auth

         // ── Design preview — somente em desenvolvimento ─────────────────────────
         if (pathname.startsWith("/design-preview") && process.env.NODE_ENV === "production") {
           return new NextResponse(null, { status: 404 })
         }

         const ip = req.headers.get("x-forwarded-for")?.split(",")[0]?.trim() ?? "unknown"

         // ── Rate limiting em endpoints de autenticação (Upstash; INFRA-01) ─────
         if (
           pathname.startsWith("/api/auth/") ||
           pathname.startsWith("/api/n8n/") ||
           pathname === "/api/user/password"
         ) {
           const { success } = await authRatelimit.limit(ip)
           if (!success) {
             return new NextResponse("Too Many Requests", { status: 429 })
           }
         }

         // ── Rate limiting em billing (Upstash; INFRA-01) ────────────────────────
         if (
           pathname === "/api/billing/checkout" ||
           pathname === "/api/billing/unified-checkout" ||
           pathname === "/api/billing/setup-charge/pay"
         ) {
           const { success } = await billingRatelimit.limit(ip)
           if (!success) {
             return new NextResponse("Too Many Requests", { status: 429 })
           }
         }

         // ── Onboarding guard — usuário sem org: só /onboarding (step 1) liberado ──
         if (session && !session.user.isSuperAdmin && !session.user.organizationId) {
           if (pathname !== "/onboarding" && pathname !== "/api/auth/onboarding") {
             return NextResponse.redirect(new URL("/onboarding", req.url))
           }
         }
         if (pathname === "/onboarding" && session?.user.organizationId) {
           return NextResponse.redirect(new URL("/dashboard", req.url))
         }
         if (pathname.startsWith("/onboarding/org-profile")) {
           if (!session) return NextResponse.redirect(new URL("/login", req.url))
           if (!session.user.organizationId) return NextResponse.redirect(new URL("/onboarding", req.url))
         }

         // ── EmailVerified gate (D-01) ───────────────────────────────────────────
         // Authenticated user, has org, has NOT verified email → /verify-email.
         // Skips bypass paths so the user can verify, resend, or sign out.
         if (
           session &&
           !session.user.isSuperAdmin &&
           session.user.organizationId &&
           session.user.emailVerified === false &&
           !isEmailVerifyBypass(pathname)
         ) {
           return NextResponse.redirect(new URL("/verify-email", req.url))
         }

         // If a verified user lands on /verify-email, send them to /dashboard.
         if (
           pathname === "/verify-email" &&
           session?.user.emailVerified === true &&
           session.user.organizationId
         ) {
           return NextResponse.redirect(new URL("/dashboard", req.url))
         }

         // ── Proteção do painel admin ────────────────────────────────────────────
         if (pathname.startsWith("/admin")) {
           if (!session) return NextResponse.redirect(new URL("/login", req.url))
           if (!session.user.isSuperAdmin) return NextResponse.redirect(new URL("/dashboard", req.url))
         }

         // ── Subscription guard ──────────────────────────────────────────────────
         const isGuardedRoute = GUARDED_PREFIXES.some((p) => pathname.startsWith(p))
         const isSettingsOrgException = pathname.startsWith("/settings/organization")
         if (isGuardedRoute && !isSettingsOrgException && session && !session.user.isSuperAdmin) {
           const status = session.user.subscriptionStatus
           if (status && BLOCKED_STATUSES.has(status)) {
             return NextResponse.redirect(new URL("/billing", req.url))
           }
         }

         return NextResponse.next()
       })

       export const config = {
         matcher: [
           "/admin/:path*",
           "/onboarding/:path*",
           "/onboarding",
           "/dashboard/:path*",
           "/flows/:path*",
           "/approvals/:path*",
           "/settings/:path*",
           "/verify-email",
           "/api/auth/:path*",

           "/api/user/:path*",
           "/api/billing/checkout",
           "/api/billing/unified-checkout",
           "/api/billing/setup-charge/pay",
         ],
       }

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Browser → proxy.ts | Every request crosses; session derived from signed JWT cookie |
| App → Upstash REST | Rate-limit increments traverse public internet |
| Browser → /verify-email | Reachable while unverified (intentional); rate-limited via authRatelimit |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-06-01 | Elevation of Privilege | Unverified user accessing /dashboard or any guarded route | mitigate | Gate: `session.user.emailVerified === false` triggers redirect to /verify-email. Claim derived server-side from DB (Plan 01 jwt callback); user cannot forge. Verified by Playwright spec 1. |
| T-06-02 | Denial of Service | Cross-instance rate-limit bypass via in-memory Map | mitigate | Replaced with Upstash atomic counters (sliding/fixed window). All Vercel instances share state. INFRA-01 closed. |
| T-06-03 | Denial of Service | Attacker spams /verify-email page (which is bypassed) | mitigate | /verify-email loads a Server Component but its API endpoints (verify-otp, resend-otp) are still inside the authRatelimit window (matcher includes /api/auth/:path*). The page itself is cheap to render. |
| T-06-04 | Spoofing | Stale JWT after verification (still has emailVerified=false) | mitigate | Plan 02 Task 3 calls `useSession().update()` after 200; that re-runs the jwt callback (Plan 01) which re-selects users.emailVerified from DB. Verified by Playwright spec 4. |
| T-06-05 | Information Disclosure | Bypass list reveals which API routes exist | accept | Bypass set is small and documented; route names are already public (404-able). No additional disclosure. |
| T-06-06 | Tampering | Attacker manipulates `?callbackUrl=` to bypass gate | mitigate | Gate evaluates `pathname` (req.nextUrl.pathname), not query params. Redirect target hard-coded to `/verify-email`. |
| T-06-07 | Repudiation | No log when gate triggers | accept | Phase 1 scope. Future: structured log per redirect for security telemetry. |
| T-06-08 | Denial of Service | Upstash outage breaks proxy.ts (every request fails) | mitigate | Plan 01 ratelimit module has dev fallback no-op when UPSTASH_REDIS_REST_URL is empty. In production, Upstash 99.99% SLA + the @upstash/ratelimit library catches errors and returns success: true on Redis failure (fail-open) — rate-limiting becomes ineffective during outage but app stays up. Risk: short DoS window during Upstash outage. Acceptable. |

**Blocking severity:** T-06-01 (unverified access), T-06-02 (rate-limit bypass), T-06-04 (stale JWT). All mitigated.
