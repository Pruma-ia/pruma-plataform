    ```typescript
    import { NextResponse } from "next/server"
    import { z } from "zod"
    import { eq } from "drizzle-orm"
    import { auth } from "@/lib/auth"
    import { db } from "@/lib/db"
    import { users } from "../../../../../db/schema"

    const schema = z.object({
      name: z
        .string()
        .min(1)
        .max(120)
        .transform((s) => s.trim())
        .refine((s) => s.length >= 1, { message: "Nome não pode estar em branco." }),
    })

    export async function PATCH(req: Request) {
      const session = await auth()
      if (!session?.user?.id) {
        return NextResponse.json({ error: "unauthorized" }, { status: 401 })
      }

      let body: unknown
      try { body = await req.json() } catch { return NextResponse.json({ error: "invalid_json" }, { status: 400 }) }
      const parsed = schema.safeParse(body)
      if (!parsed.success) {
        return NextResponse.json({ error: parsed.error.flatten() }, { status: 400 })
      }

      await db
        .update(users)
        .set({ name: parsed.data.name })
        .where(eq(users.id, session.user.id))

      return NextResponse.json({ ok: true })
    }
    ```

       ```typescript
       export interface ConnectedAccount {
         provider: string
         providerAccountId: string
       }

       export async function getConnectedAccounts(userId: string): Promise<ConnectedAccount[]> {
         const [oauthRows, [userRow]] = await Promise.all([
           db
             .select({ provider: accounts.provider, providerAccountId: accounts.providerAccountId })
       ```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Browser → /api/user/profile | Untrusted name string crosses |
| Browser → /settings/profile (read) | Server component reads from session.user.id only |
| Server → accounts table | Internal query, scoped by userId from session |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-05-01 | Tampering | User submits arbitrary userId in PATCH body | mitigate | Route reads userId only from `session.user.id`; body schema does not declare a userId field. Drizzle update scoped to that id. |
| T-05-02 | Information Disclosure | Cross-user accounts leak | mitigate | `getConnectedAccounts(userId)` filters by `eq(accounts.userId, userId)`; `userId` comes from session in the page server component. |
| T-05-03 | Denial of Service | PATCH /api/user/profile spam | mitigate | Already covered by `proxy.ts` `/api/user/*` rate limit (now Upstash via Plan 06). Per-user impact only. |
| T-05-04 | XSS | name field rendered in dashboard layouts (header personalization) | mitigate | name is rendered via React text interpolation (auto-escaped). No dangerouslySetInnerHTML anywhere. Zod max length 120 caps payload size. |
| T-05-05 | Repudiation | No audit trail for name changes | accept | Phase 1 scope. Future enhancement: audit log row per name change. |
| T-05-06 | Elevation of Privilege | User attempts to disconnect last auth method | mitigate | Disconnect intentionally NOT implemented in Phase 1 (locked CONTEXT discretion). No code path exists to remove a provider, so the "last method" footgun cannot be triggered here. |

**Blocking severity:** T-05-01 (userId tampering), T-05-02 (cross-user leak). Both mitigated.
